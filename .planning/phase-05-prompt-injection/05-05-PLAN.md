    def _parse_prompt_injection(form):
        errors = {}
        raw_patterns = _lines_to_list(form.get("prompt_injection.regex_patterns", ""))
        invalid_idx = []
        for i, p in enumerate(raw_patterns, start=1):
            try:
                compiled = re.compile(p)
            except re.error:
                invalid_idx.append((i, p))
                continue
            # ReDoS probe
            if not _redos_safe(compiled):
                invalid_idx.append((i, p))
        if invalid_idx:
            i, p = invalid_idx[0]
            errors["prompt_injection"] = f"Невалидный regex в строке {i}: «{p}». Исправьте и сохраните снова."
            return ({...preserve_raw...}, errors)

        raw_allow = _lines_to_list(form.get("prompt_injection.allowlist_patterns", ""))
        for i, p in enumerate(raw_allow, start=1):
            check_pattern = p[3:] if p.startswith("re:") else None
            if check_pattern is not None:
                try:
                    re.compile(check_pattern)
                except re.error:
                    errors["prompt_injection"] = f"Невалидный regex в allowlist, строка {i}: «{p}»."
                    return (..., errors)

        severity = form.get("prompt_injection.severity", "warn")
        if severity not in ("info", "warn", "block"):
            errors["prompt_injection"] = "severity должен быть одним из: info, warn, block."
            return (..., errors)

        lg_enabled = _checkbox(form.get("prompt_injection.llama_guard.enabled", ""))
        lg_endpoint = form.get("prompt_injection.llama_guard.endpoint", "http://localhost:11434").strip()
        if lg_enabled and not _is_valid_url(lg_endpoint):
            errors["prompt_injection"] = "Endpoint LlamaGuard должен быть валидным URL (http:// или https://)."
            return (..., errors)

        try:
            timeout_ms = int(form.get("prompt_injection.llama_guard.timeout_ms", "500"))
            if not (50 <= timeout_ms <= 10000):
                raise ValueError()
        except ValueError:
            errors["prompt_injection"] = "timeout_ms должен быть в диапазоне 50–10000 мс."
            return (..., errors)

        return ({
            "enabled": _checkbox(form.get("prompt_injection.enabled", "")),
            "severity": severity,
            "regex_patterns": raw_patterns,
            "allowlist_patterns": raw_allow,

    Helpers:
    ```
    def _is_valid_url(s: str) -> bool:
        return s.startswith("http://") or s.startswith("https://")

    def _redos_safe(compiled: re.Pattern, probe: str = "a" * 1000, budget_ms: int = 50) -> bool:
        from concurrent.futures import ThreadPoolExecutor, TimeoutError
        with ThreadPoolExecutor(max_workers=1) as ex:
            fut = ex.submit(compiled.search, probe)
            try:
                fut.result(timeout=budget_ms / 1000.0)
                return True
            except TimeoutError:
                return False
    ```
