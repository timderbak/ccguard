         def anomalies_overview_partial(request: Request, session: Session = Depends(get_session), _user = Depends(require_admin)):
             rows = session.exec(
                 select(FindingRecord)
                 .where(FindingRecord.rule_id.like("anomaly.%"))
                 .order_by(FindingRecord.discovered_at.desc())
                 .limit(5)
             ).all()
             items = []
             for r in rows:
                 metric = r.rule_id.removeprefix("anomaly.")
                 payload = json.loads(r.payload_json) if r.payload_json else {}
                 items.append({
                     "machine_id": r.machine_id,
                     "metric": metric,
                     "observed_value": payload.get("observed_value", "—"),
                     "sigma_distance": round(payload.get("sigma_distance", 0.0), 1),
                     "ts_short": r.discovered_at.strftime("%Y-%m-%d %H:%M"),
                 })
             return templates.TemplateResponse("components/_anomalies_overview.html", {"request": request, "items": items})
